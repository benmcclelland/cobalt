# Command Line Options and Arguments to test cqadm

#
# get queues option
#
--getq                              # <tc:getq_option_1> <rs:0>
-d --getq                           # <tc:getq_option_2> <rs:0>
-f --getq                           # <tc:getq_option_3> <rs:0>

#
# preempt jobs option
#
-d --preempt 1 2 3                  # <tc:preempt_job_1> <rs:0>
-f --preempt 1 2 3                  # <tc:preempt_job_2> <rs:0>

#
#  delete jobs options
#
--delete 1 2 3 | -d -f --kill 1 2 3 # <tc:kill_job_1> <rs:0>
--kill 1 2 3                        # <tc:kill_job_2> <rs:0>
-d --delete 1 2 3 | -f --kill 1 2 3 # <tc:kill_job_3> <rs:0>
-d --kill 1 2 3                     # <tc:kill_job_4> <rs:0>

#
# set queues options
#
|--addq                             # <tc:addq_option_1> <rs:256>
--addq myq1 myq2 myq3               # <tc:addq option_2> <rs:0>
|--delq                             # <tc:delq_option_1> <rs:256>
--delq myq1 myq2 myq3               # <tc:delq option_2> <rs:0>
|--stopq                            # <tc:stopq_option_1> <rs:256>
--stopq myq1 myq2 myq3              # <tc:stopq_option_2> <rs:0>
|--startq                           # <tc:startq_option_1> <rs:256>
--startq myq1 myq2 myq3             # <tc:startq_option_2> <rs:0>
|--drainq                           # <tc:drainq_option_1> <rs:256>
--drainq myq1 myq2 myq3             # <tc:drainq_option_2> <rs:0>
|--killq                            # <tc:killq_option_1> <rs:256>
--killq myq1 myq2 myq3              # <tc:killq_option_2> <rs:0>
--policy                            # <tc:policy_option_1> <rs:512>
|--policy 'mypolicy'                # <tc:policy_option_2> <rs:256>
--policy 'mypolicy' myq1 myq2       # <tc:policy_option_3> <rs:0>
--setq                              # <tc:setq_option_1> <rs:512>
|--setq 'a=b b=c a=c'               # <tc:setq_option_2> <rs:256>
--setq 'a=b b=c a=c' myq1 myq2      # <tc:setq_option_3> <rs:0>
--unsetq                            # <tc:unsetq_option_1> <rs:512>
|--unsetq 'a b a'                   # <tc:unsetq_option_2> <rs:256>
--unsetq 'a b a' myq1 myq2          # <tc:unsetq_option_3> <rs:0>

#
# setjobid options
#
-j                                  # <tc:setjobid_option_1> <rs:512>
--setjobid                          # <tc:setjobid_option_2> <rs:512>

--setjobid 1                        # <tc:setjobid_option_4> <rs:0>
-j 1 --setjobid 2                   # <tc:setjobid_option_5> <rs:0>

#
# run job option
#
--run                               # <tc:run_option_1> <rs:512>
|--run mayaguez                     # <tc:run_option_2> <rs:256>
--run mayaguez 1 2 3                # <tc:run_option_3> <rs:0>
#
# set jobs options
#
--hold                              # <tc:hold_option_1> <rs:256>
--hold 1 2 3                        # <tc:hold_option_2> <rs:0>
-d --hold  1 2 3                    # <tc:hold_option_3> <rs:0>
-f --hold  1 2 3                    # <tc:hold_option_4> <rs:0>
--release                           # <tc:release_option_1> <rs:256>
--release 1 2 3                     # <tc:release_option_2> <rs:0>
-d --release 1 2 3                  # <tc:release_option_3> <rs:0>
-f --release 1 2 3                  # <tc:release_option_4> <rs:0>

|--queue myq                        # <tc:queue_option_2> <rs:256>

--hold --queue myq --time 50 4 5 6    # <tc:update_all_1> <rs:0>
